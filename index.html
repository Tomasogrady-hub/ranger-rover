const SITES_ID  = '1fs9T_fhevN-6_NgaDV941-RaQMC5mF52yc8eDitgsJc';
const HUMANS_ID = '19s6gQeFJWeVcAezkE1Lg6MJcZ3C8-h-IdWmZUDT5cHk';
const BLOCKED_H = ['Comments','Rating 1','Rating 2','Rating 3'];
const BLOCKED_S = ['Comments'];

function doGet() {
  var sites  = readSheet(SITES_ID,  'Sites',  BLOCKED_S);
  var humans = readSheet(HUMANS_ID, 'Humans', BLOCKED_H);
  return respond({sites:sites, humans:humans});
}

function doPost(e) {
  try {
    var p = JSON.parse(e.postData.contents);
    var id  = p.sheet==='Humans' ? HUMANS_ID : SITES_ID;
    var tab = p.sheet==='Humans' ? 'Humans'  : 'Sites';
    var keyCol = p.sheet==='Humans' ? 'Email' : 'Name';
    var sheet = SpreadsheetApp.openById(id).getSheetByName(tab);
    var data = sheet.getDataRange().getValues();
    var headers = data[0];
    var ki = headers.indexOf(keyCol);
    for (var r = 1; r < data.length; r++) {
      if (String(data[r][ki]) === String(p.key)) {
        Object.keys(p.updates).forEach(function(col) {
          var ci = headers.indexOf(col);
          if (ci > -1) sheet.getRange(r+1,ci+1).setValue(p.updates[col]);
        });
        return respond({ok:true});
      }
    }
    return respond({ok:false, error:'Row not found'});
  } catch(err) {
    return respond({ok:false, error:err.message});
  }
}

function readSheet(id, tab, blocked) {
  var data = SpreadsheetApp.openById(id)
    .getSheetByName(tab).getDataRange().getValues();
  var h = data[0];
  return data.slice(1).map(function(row) {
    var obj = {};
    h.forEach(function(k,i){ if(blocked.indexOf(k)===-1) obj[k]=row[i]; });
    return obj;
  });
}

function respond(obj) {
  return ContentService
    .createTextOutput(JSON.stringify(obj))
    .setMimeType(ContentService.MimeType.JSON);
}
