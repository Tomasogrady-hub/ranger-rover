GAS Script Template

Add this doGet() to your Apps Script, deploy as web app (Anyone): 

const SITES_ID  = '1fs9T_fhevN-6_NgaDV941-RaQMC5mF52yc8eDitgsJc'; // Ranger Rover
const HUMANS_ID = '19s6gQeFJWeVcAezkE1Lg6MJcZ3C8-h-IdWmZUDT5cHk'; // Humans

function doGet() {
  var sites  = sheetToObjects(SITES_ID,  'Sites');   // tab: Sites
  var humans = sheetToObjects(HUMANS_ID, 'Humans'); // tab: Humans
  var out = ContentService.createTextOutput(
    JSON.stringify({sites:sites, humans:humans})
  );
  out.setMimeType(ContentService.MimeType.JSON);
  return out;
}

function sheetToObjects(id, sheetName) {
  var sheet = SpreadsheetApp
    .openById(id).getSheetByName(sheetName);
  var data  = sheet.getDataRange().getValues();
  var headers = data[0];
  return data.slice(1).map(function(row) {
    var obj = {};
    headers.forEach(function(h,i){obj[h]=row[i];});
    return obj;
  });
}
// Deploy: Web App > Anyone > Execute as Me
